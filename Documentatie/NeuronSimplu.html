Examplu de calcul al functiei sigmoid

 

function sigmoid (v, g: double):double;

begin

     result := 1 / (1 + exp(0 - (g*v)));

end;

 

 

Functia tanh

 

function TForm1.tanh(v:double):double;

begin

   result := (exp(v)- exp(0-v))/(exp(v)+exp(0-v))

end;

 

Functia Treapta

function treshold(v:double):double;

begin

  if v >= 0 then

     result :=1

  else

     result :=0;

 

end;

 

Functia Lineara

function Linear(a,v:double):double;

begin

  result := v*a;

 

end;

Functia Semn

function signum(v:double):double;

begin

  if v >= 0 then

     result :=1

  else

     result :=-1;

end;

Functia Rampa

function rampa(v:double):double;

begin

  if v >= 1 then

     result :=1

  else

     if v < -1 then

     result :=-1

     else

     result := v;

end;
